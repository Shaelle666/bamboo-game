import tkinter as tk import math import time # ----------------------------- # 十年孤竹 - 猎杀时刻（Tkinter版 / Python 3.14 可直接运行） # 交互： # - 第一次：点击竹子 -> 竹子断裂 + 出魂环 + 匕首跟随 # - 第二次：点击任意位置 -> 魂环消失 + 匕首消失 + 显示“你已吸收魂环，魂技：撑杆跳” # ----------------------------- W, H = 800, 600 WHITE = "#FFFFFF" GREEN = "#228B22" BROWN = "#8B4513" BLACK = "#000000" YELLOW = "#FFFF00" # 状态：0=竹子活着；1=魂环出现；2=已吸收 state = 0 # 竹子区域 bamboo_x, bamboo_y, bamboo_w, bamboo_h = 380, 200, 40, 300 # 鼠标位置（匕首跟随） mx, my = W // 2, H // 2 # 计时 start_time = time.time() root = tk.Tk() root.title("十年孤竹 - 猎杀时刻 (Tkinter)") root.resizable(False, False) canvas = tk.Canvas(root, width=W, height=H, bg=BLACK, highlightthickness=0) canvas.pack() def in_bamboo(x, y): return bamboo_x <= x <= bamboo_x + bamboo_w and bamboo_y <= y <= bamboo_y + bamboo_h def on_move(e): global mx, my mx, my = e.x, e.y def on_click(e): global state if state == 0: if in_bamboo(e.x, e.y): state = 1 # 砍断 + 出魂环 elif state == 1: state = 2 # 吸收（点任意地方） canvas.bind("", on_move) canvas.bind("", on_click) def draw_dagger(x, y): # 三角形匕首 canvas.create_polygon( x, y, x + 10, y + 30, x - 10, y + 30, fill=WHITE, outline="" ) def draw(): canvas.delete("all") # 背景 canvas.create_rectangle(0, 0, W, H, fill=BLACK, outline="") # 竹子 / 残骸 if state == 0: # 竹子 canvas.create_rectangle( bamboo_x, bamboo_y, bamboo_x + bamboo_w, bamboo_y + bamboo_h, fill=GREEN, outline="" ) # 竹节 for i in range(5): y = 250 + i * 50 canvas.create_line(bamboo_x, y, bamboo_x + bamboo_w, y, fill=BLACK, width=2) else: # 残骸底座（和你原来差不多） canvas.create_rectangle(380, 450, 420, 500, fill=BROWN, outline="") # 魂环（只在 state==1 显示） if state == 1: t = time.time() - start_time offset = math.sin(t * 3.0) * 10 cx, cy = 400, 350 + int(offset) # 外圈 canvas.create_oval(cx - 50, cy - 20, cx + 50, cy + 20, outline=WHITE, width=3) # 内圈（微粒感） canvas.create_oval(cx - 45, cy - 15, cx + 45, cy + 15, outline=WHITE, width=1) # 吸收成功文字（state==2） if state == 2: canvas.create_text( 400, 300, text="你已吸收魂环，魂技：撑杆跳", fill=YELLOW, font=("SimHei", 28) ) # 匕首（只在未吸收时显示：state 0 或 1） if state in (0, 1): draw_dagger(mx, my) root.after(16, draw) # ~60fps draw() root.mainloop()
